# Colors

Colors play a crucial role in creating visually appealing and consistent user interfaces.


<h2 id="color-palette">Color palette</h2>

The neetoUI color palette is carefully curated to provide a harmonious and visually appealing experience. It consists of primary, gray, status and pastel colors.
<br />

<h3 id="primary-color">Primary</h3>

The primary color is the most dominant color in our brand identity.

<InlineComponent>
  {() => {
    const isDarkMode = useDarkMode();
    const theme = isDarkMode ? "dark" : "light";
    return (
      <table className="story-demo-table">
        <thead>
        <tr>
          <td style={{ width: "75px" }}></td>
          <td>Color</td>
          <td>Use for</td>
        </tr>
        </thead>
        <tbody>
        {getColorPaletteList("PRIMARY", theme)
        .filter((color) => color.usage !== undefined)
        .map((color) => {
          return (
            <tr key={color.name}>
              <td>
                <div
                  className={`w-12 h-12 cursor-pointer rounded-full shadow-xl ${color.name.replace(
                    "--neeto-ui",
                    "neeto-ui-bg"
                  )}`}
                ></div>
              </td>
              <td>
                <code>{`${color.name}`}</code>
              </td>
              <td>
                {`${color.usage}`}
              </td>
            </tr>
          );
        })}
        </tbody>
      </table>
    );
  }}
</InlineComponent>

<h2 id="neeto-ui-colors-cheatsheet">Colors cheatsheet</h2>
<br/>

<InlineComponent>
  {() => {
    const isDarkMode = useDarkMode();
    const theme = isDarkMode ? "dark" : "light";
    return (
      <table className="story-demo-table">
        <thead>
        <tr>
          <td>RGB code</td>
          <td>Hex code</td>
          <td>CSS Variable name</td>
          <td>Text color class</td>
          <td>Background color class</td>
        </tr>
        </thead>
        <tbody>
        {colorPaletteList(theme).map((color) => {
          return (
            <tr key={color.name}>
              <td>
                 <code>{color.value}</code>
              </td>
              <td>
                  <code>#{tinycolor(`rgb(${color.value})`).toHex()}</code>
              </td>
              <td>
                <code>{`${color.name}`}</code>
              </td>
              <td>
                <div className="flex justify-between">
                  <code>{`${color.name.replace(
                    "--neeto-ui",
                    "neeto-ui-text"
                  )}`}</code>
                  <b
                    className={`ml-2 ${color.name.replace(
                      "--neeto-ui",
                      "neeto-ui-text"
                    )}`}
                  >
                    Aa
                  </b>
                </div>
              </td>
              <td
                className={classnames(
                  color.name.replace("--neeto-ui", "neeto-ui-bg"),
                )}
              >
                <div
                  className={classnames(`neeto-ui-text-sm`,
                    {
                      "neeto-ui-text-black": color.text === "black",
                      "neeto-ui-text-white": color.text !== "black",
                    }
                  )}
                >
                  {`${color.name.replace("--neeto-ui", "neeto-ui-bg")}`}
                </div>
              </td>
            </tr>
          );
        })}
        </tbody>
      </table>
    );
  }}
</InlineComponent>
